infix 9 _⊩_
data _⊩_ : Context → Type → Set where
  rec : ∀{Γ τ₁ τ₂}
      (t : Γ , τ₁ ⇒ τ₂ ⊢ τ₁ ⇒ τ₂)
    → (τ₁ ⇒ τ₂) called-in t
    ----------------
    → Γ ⊩ τ₁ ⇒ τ₂

  rec∙ : ∀{Γ τ₁ τ₂}
    → Γ ⊩ τ₁ ⇒ τ₂
    → Γ ⊢ τ₁
    --------------
    → Γ ⊩ τ₂
